x= int(input())
st=input()
count=0
for y in range(0,x):
    if st[y]==st[0]:
        y+=1
    else:
        break
if len(st)==x:
    for i in range(y,x+1):
        if st[i-1]!=st[y]:
            n=st[i]
            if st[i+1]!=st[i]:
                count+=1
            if st[i+1]==n:
                break
else:
    print('length not sufficiant')
print(count)
